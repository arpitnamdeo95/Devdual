import numpy as np

class KMeans:
    def __init__(self, k=3, max_iter=300):
        self.k = k
        self.max_iter = max_iter
        self.centroids = None

    def fit(self, X):
        # Initialize centroids using K-Means++
        self.centroids = [X[np.random.choice(X.shape[0])]]
        for _ in range(1, self.k):
            dist_sq = np.array([min([np.inner(c-x, c-x) for c in self.centroids]) for x in X])
            probs = dist_sq / dist_sq.sum()
            cumulative_probs = probs.cumsum()
